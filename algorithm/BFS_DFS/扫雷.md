class Solution:
    def updateBoard(self, board: List[List[str]], click: List[int]) -> List[List[str]]:
        from collections import deque
        nrows,ncols = len(board),len(board[0])
        row,col = click

        if board[row][col] == "M":
            board[row][col] = "X"
            return board

        def isValid(row,col):
            return 0<=row< nrows and 0 <= col < ncols

        def cal_nums(r,c):
            res = 0
            for i in range(-1,2):
                for j in range(-1,2):
                    if not isValid(r+i,c+j):
                        continue
                    if board[r+i][c+j] == "M":
                        res += 1
            return res

        def bfs(row,col):
            queue = deque([(row,col)])
            while queue:
                for _ in range(len(queue)):
                    r,c = queue.pop()
                    # 计算雷的个数
                    nums = cal_nums(r,c)
                    if nums > 0:
                        board[r][c] = str(nums)
                    else:
                        #沉岛
                        board[row][col] = "B"
                        for i in range(-1,2):
                            for j in range(-1,2):
                                if not isValid(r+i,c+j):
                                    continue
                                if board[r+i][c+j] == "E":
                                    #下一层
                                    queue.appendleft((r+i,c+j))
                                    board[r+i][c+j] = "B"
        bfs(row,col)
        return board
